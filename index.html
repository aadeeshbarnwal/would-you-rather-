import random # A predefined pool of starter questions HAND_CRAFTED_QUESTIONS = [ ("have the ability to fly", "be able to teleport anywhere instantly"), ("always be 15 minutes late", "always be 20 minutes early"), ("find true love", "win 10 million dollars"), ("lose the ability to speak", "always say everything that is on your mind"), ("live without music", "live without television"), ("be an unknown Olympic athlete", "be a famous billionaire who sits on the couch"), ("know the history of every object you touch", "be able to talk to animals"), ] # dynamic word lists to procedurally generate infinite combinations later SUPERPOWERS = ["invisible", "immortal", "able to breathe underwater", "able to read minds", "super strong", "able to change your appearance"] CURSES = ["stub your toe every morning", "only eat cold soup", "have your phone always at 1%", "always hear a faint buzzing", "forget everyone's name"] REWARDS = ["free pizza for life", "never needing sleep", "mastering every musical instrument", "perfect health forever"] def generate_dynamic_question(): """Generates a structurally unique scenario dynamically.""" templates = [ ("be completely {power_1}", "be completely {power_2}"), ("deal with having to {curse_1}", "deal with having to {curse_2}"), ("get {reward_1}", "get {reward_2}"), ("be {power_1} but you {curse_1}", "be ordinary but you get {reward_1}") ] template = random.choice(templates) # Select unique items per structure to avoid nonsense duplicates within a single question p1, p2 = random.sample(SUPERPOWERS, 2) c1, c2 = random.sample(CURSES, 2) r1, r2 = random.sample(REWARDS, 2) option_a = template[0].format(power_1=p1, power_2=p2, curse_1=c1, curse_2=c2, reward_1=r1, reward_2=r2) option_b = template[1].format(power_1=p1, power_2=p2, curse_1=c1, curse_2=c2, reward_1=r1, reward_2=r2) return option_a, option_b def get_next_question(asked_set, fixed_pool): """Fetches from the static pool first, then generates dynamic alternatives.""" # Attempt to use hand-crafted pool first available_fixed = [q for q in fixed_pool if q not in asked_set] if available_fixed: chosen = random.choice(available_fixed) asked_set.add(chosen) return chosen # fallback to creating dynamic questions indefinitely while True: dynamic_q = generate_dynamic_question() if dynamic_q not in asked_set: asked_set.add(dynamic_q) return dynamic_q def play_game(): asked_questions = set() # Tracks history globally to prevent repeats print("=" * 50) print("Welcome to the Ultimate UNLIMITED 'Would You Rather' Game!") print("=" * 50) q_count = 1 while True: option_a, option_b = get_next_question(asked_questions, HAND_CRAFTED_QUESTIONS) print(f"\n[Question {q_count}] Would you rather...") print(f"Option (A): {option_a}") print(f"Option (B): {option_b}") # Input Validation Loop while True: choice = input("Your choice (A/B) or type 'exit' to quit: ").strip().lower() if choice in ['a', 'b', 'exit']: break print("Invalid selection. Please choose 'A', 'B', or type 'exit'.") if choice == 'exit': print(f"\nThanks for playing! You answered {q_count - 1} questions without any repeats.") break print(f"-> You chose Option {choice.upper()}!") q_count += 1 if __name__ == "__main__": play_game()
